Var		g_Dur				
Var		g_Step				
Func Loop
{
	Mov		g_Dur, 0			
	Mov		g_Step, 10			
	LoopStart:					
		Add		g_Dur, g_Step		
		JGE		g_Dur, 400, Negate	
		JLE		g_Dur, 0, Negate	
		Jmp		Skip			
	Negate:
		Neg		g_Step
	Skip:
		Pause		g_Dur			
	Jmp		LoopStart			
}
Func _Main						
{							
	Call		Loop				
}
